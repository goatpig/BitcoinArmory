#compile capnp schemas
SET(CAPNP_FILES ${PROJECT_BINARY_DIR}/cppForSwig/capnp)
FILE(MAKE_DIRECTORY ${CAPNP_FILES})

FOREACH(schema ${CAPNPROTO_FILES})
    SET(CAPNP_CC "${schema}.c++")
    ADD_CUSTOM_COMMAND(
        OUTPUT ${CAPNP_FILES}/${CAPNP_CC}
        COMMAND ${CAPNP_EXECUTABLE} compile --no-standard-import --import-path=${CAPNP_INCLUDE_DIRS} --src-prefix=${CMAKE_CURRENT_SOURCE_DIR}/capnp --output=${CAPNP_PATH}/build/${CAPNP_SUFFIX}/capnp/capnpc-c++.exe\:capnp ${CMAKE_CURRENT_SOURCE_DIR}/capnp/${schema}
    )
    LIST(APPEND LIBARMORYCOMMON_SOURCES ${CAPNP_FILES}/${CAPNP_CC})
ENDFOREACH()
